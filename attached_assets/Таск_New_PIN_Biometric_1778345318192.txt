Правильна архітектура

Логіка розблокування:
  1. isPINSet()? НІ → додаток відкривається без блоку
  2. isPINSet()? ТАК → показати lock screen
     → biometricEnabled + biometricAvailable?
         ТАК → одразу запустити authenticate() (системний діалог)
               → успіх → unlock
               → відмова/помилка → показати наш PIN-пад
         НІ  → одразу показати PIN-пад
		 


Задача: Додати розблокування відбитком як опцію поверх PIN

Файл 1 — AndroidManifest.xml — додати 2 рядки після ACTIVITY_RECOGNITION:

<!-- Biometric authentication (fingerprint / face unlock) -->
<uses-permission android:name="android.permission.USE_BIOMETRIC" />
<uses-permission android:name="android.permission.USE_FINGERPRINT" />

Файл 2 — src/core/biometric.js — замінити повністю:

let _BiometricAuth = null;

async function _loadPlugin() {
  if (_BiometricAuth !== null) return _BiometricAuth;
  try {
    const m = await import('@aparajita/capacitor-biometric-auth');
    _BiometricAuth = m.BiometricAuth ?? null;
  } catch {
    _BiometricAuth = false;
  }
  return _BiometricAuth || null;
}

export async function checkBiometric() {
  try {
    const plugin = await _loadPlugin();
    if (!plugin) return false;
    const info = await plugin.checkBiometry();
    // Тільки відбиток/обличчя — НЕ системний PIN
    return !!(info?.isAvailable);
  } catch { return false; }
}

export async function authenticate() {
  try {
    const plugin = await _loadPlugin();
    if (!plugin) return false;
    await plugin.authenticate({
      reason: 'Підтвердіть особу для доступу до HealthPro',
      cancelTitle: 'Використати PIN',
      allowDeviceCredential: false, // ← НЕ змішувати з нашим PIN
      iosFallbackTitle: 'Використати PIN',
    });
    return true;
  } catch {
    return false; // скасування → fallback на PIN-пад у app.js
  }
}

Файл 3 — app.js — знайти функцію lockCheck (або init блокування) і замінити логіку:

// При старті додатку — якщо PIN встановлено:
async function lockCheck() {
  if (!isPINSet()) return; // блокування вимкнено

  const bioEnabled   = state.settings.biometricEnabled ?? false;
  const bioAvailable = await checkBiometric();

  if (bioEnabled && bioAvailable) {
    // Спочатку пробуємо відбиток
    const ok = await authenticate();
    if (ok) {
      unlockApp();
      return;
    }
    // Відмова або помилка → плавно показуємо PIN-пад
  }
  showLockScreen(); // показати наш PIN-пад
}

Файл 6 — app.js actions — додати/оновити toggleBiometric:

case 'toggleBiometric': {
  const enabled = e.target.checked;
  // Не дозволяти вмикати якщо немає PIN
  if (enabled && !isPINSet()) {
    e.target.checked = false;
    showToast(t('bio-toggle-hint'));
    return;
  }
  state.settings.biometricEnabled = enabled;
  saveData();
  break;
}

Ключові відмінності від попередньої спроби:

allowDeviceCredential: false — тільки відбиток, без системного PIN
checkBiometry() перевіряє isAvailable (відбиток) — не deviceIsSecure (системний PIN)
USE_BIOMETRIC + USE_FINGERPRINT дозволи в маніфесті — це було відсутнє
Fallback на PIN-пад при будь-якій помилці плагіну
